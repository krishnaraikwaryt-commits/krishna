const UI = { navigate: (view) => { // 1. Saare sections ko hide karo (Includng alerts) const sections = ['dashboard', 'leads', 'clients', 'alerts']; sections.forEach(s => { const el = document.getElementById(`view-${s}`); if(el) el.classList.add('hidden'); }); const title = document.getElementById('view-title'); const today = new Date().toISOString().split('T')[0]; // 2. Alert Logic Fix if(view.startsWith('alerts')) { document.getElementById('view-alerts').classList.remove('hidden'); let filtered = []; if(view === 'alerts-today') { title.innerText = "Today's Tasks"; filtered = System.Data.filter(l => l.followup === today); } else if(view === 'alerts-overdue') { title.innerText = "Overdue Actions"; // Filter: Date purani honi chahiye aur status Closed nahi hona chahiye filtered = System.Data.filter(l => l.followup && l.followup < today && l.status !== 'Closed Won'); } // Yahan data inject ho raha hai document.getElementById('view-alerts').innerHTML = `

Found: ${filtered.length} Leads

${UI.cards(filtered)} `; } else { // Standard Navigation document.getElementById(`view-${view}`).classList.remove('hidden'); title.innerText = view.charAt(0).toUpperCase() + view.slice(1); } // Dropdown band karo aur top par scroll karo document.getElementById('theme-menu').classList.remove('active'); window.scrollTo(0,0); }, // ... baaki UI functions same rahenge }
